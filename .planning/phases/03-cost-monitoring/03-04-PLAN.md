    5. Add CSV export button inside the Cost Breakdown card header (right of the Select):
       The export uses the blob download pattern from the research document — use the existing `api` singleton directly (NOT a hook — this is a one-time action, not server state):
       ```tsx
       import api from "@/services/api";

       const handleExport = async () => {
         setIsExporting(true);
         try {
           const response = await api.get("/costs/export", {
             params: { dimension, days },
             responseType: "blob",
           });
           const url = window.URL.createObjectURL(new Blob([response.data]));
           const link = document.createElement("a");
           link.href = url;
           link.setAttribute("download", `cost-breakdown-${dimension}-${days}d.csv`);
           document.body.appendChild(link);
           link.click();
           link.remove();
           window.URL.revokeObjectURL(url);
         } finally {
           setIsExporting(false);
         }
       };

       <Button variant="outline" size="sm" onClick={handleExport} disabled={isExporting}>
         {isExporting ? "Exporting..." : "Export CSV"}
       </Button>
       ```
